function checkAndGenerate() {
        let errors = [];
        let missing = [];
        
        // Reset alle rode randen van de vorige check
        document.querySelectorAll('.card').forEach(c => c.classList.remove('card-error'));

        questions.forEach(q => {
            const card = document.getElementById(`card_${q.id}`);
            // We controleren alleen vragen die niet verborgen zijn en niet optioneel zijn
            if (card && !card.classList.contains('hidden') && !q.isOptional) {
                
                // 1. Check of de hoofdvraag is ingevuld
                if (!formData[q.id]) {
                    missing.push(q.label);
                    card.classList.add('card-error');
                } 
                
                // 2. Check of het bouwjaar binnen de marge valt
                if (q.id === "bouwjaar" && formData[q.id]) {
                    const bj = parseInt(formData[q.id]);
                    if (bj < 500 || bj > 2100) { 
                        errors.push("Bouwjaar moet tussen 500 en 2100 liggen."); 
                        card.classList.add('card-error'); 
                    }
                }
                
                // 3. Check of GBO m2 is ingevuld als 'Gemeenschappelijk' op 'Ja' staat
                if (q.isGemeenschappelijk && formData[q.id + '_gem'] === 'Ja' && !formData[q.id + '_gem_opp']) {
                    errors.push(`${q.label}: GBO m2 is verplicht bij gemeenschappelijke installatie.`);
                    card.classList.add('card-error');
                }
            }
        });

        // Als er fouten of ontbrekende vragen zijn, toon dan een keuze-venster
        if (missing.length > 0 || errors.length > 0) {
            let msg = "Let op: Er zijn onvolledigheden gevonden:\n\n";
            
            if (missing.length > 0) {
                msg += "Niet ingevuld:\n- " + missing.join("\n- ") + "\n\n";
            }
            if (errors.length > 0) {
                msg += "Fouten:\n- " + errors.join("\n- ") + "\n\n";
            }
            
            msg += "Wil je toch doorgaan met opslaan?";
            
            // confirm() geeft een OK (true) of Annuleren (false) knop
            if (!confirm(msg)) {
                return; // Stop de functie hier als de gebruiker op 'Annuleren' klikt
            }
        }
        
        // Als alles goed is OF de gebruiker heeft op OK geklikt bij de waarschuwing:
        generateWord();
    }

    function handleImage(id, input) { Array.from(input.files).forEach(f => processFile(f, id)); }
    function handleDrop(e, id) { e.preventDefault(); Array.from(e.dataTransfer.files).forEach(f => processFile(f, id)); }
    function processFile(file, id) {
        const r = new FileReader();
        r.onload = (e) => {
            const img = new Image();
            img.src = e.target.result;
            img.onload = () => {
                const c = document.createElement('canvas');
                c.width = 800; c.height = img.height * (800 / img.width);
                c.getContext('2d').drawImage(img, 0, 0, c.width, c.height);
                if (!photoData[id]) photoData[id] = [];
                photoData[id].push(c.toDataURL('image/jpeg', 0.7));
                renderGallery(id);
            }
        }
        r.readAsDataURL(file);
    }
    function renderGallery(id) {
        const g = document.getElementById(`gallery_${id}`);
        g.innerHTML = "";
        (photoData[id] || []).forEach((img, i) => {
            const d = document.createElement('div');
            d.className = "photo-item";
            d.innerHTML = `<img src="${img}"><button class="delete-btn" onclick="photoData['${id}'].splice(${i},1);renderGallery('${id}')">&times;</button>`;
            g.appendChild(d);
        });
    }
async function generateWord() {
        // Let op: 'AlignmentType' is toegevoegd aan de lijst hieronder
        const { Document, Packer, Paragraph, TextRun, HeadingLevel, ImageRun, AlignmentType } = docx;
        const children = [];

        // --- NIEUW: LOGO TOEVOEGEN AAN WORD DOCUMENT ---
        try {
            // De link naar jouw logo
            const logoUrl = "https://de-energieurs.nl/wp-content/uploads/2023/07/logo_green.png";
            const response = await fetch(logoUrl);
            const blob = await response.blob();
            const buffer = await blob.arrayBuffer();

            // Voeg het logo toe aan het begin van het document
            children.push(new Paragraph({
                children: [
                    new ImageRun({
                        data: buffer,
                        transformation: { width: 200, height: 65 } // Breedte en hoogte in pixels
                    })
                ],
                alignment: AlignmentType.CENTER // Zet het logo in het midden
            }));
            
            // Een witregel onder het logo
            children.push(new Paragraph({ text: "" }));
            
        } catch (error) {
            console.warn("Logo kon niet worden geladen (mogelijk door serverbeveiliging/CORS).", error);
        }
        // --- EINDE LOGO ---

        // De hoofdtitel van het rapport
        children.push(new Paragraph({ text: "Woningopname Rapport", heading: HeadingLevel.HEADING_1 }));
        
        // Vanaf hier gaat de code weer verder zoals je gewend bent:
        for (const q of questions) {            const card = document.getElementById(`card_${q.id}`);
            if (!card || card.classList.contains('hidden')) continue;
            children.push(new Paragraph({ children: [new TextRun({ text: q.label, bold: true, color: "2E75B6" })], spacing: { before: 200 } }));
            children.push(new Paragraph({ text: `Antwoord: ${formData[q.id] || "-"}` }));
            if (formData[q.id + '_gem']) children.push(new Paragraph({ text: `Gemeenschappelijk: ${formData[q.id + '_gem']} ${formData[q.id + '_gem_opp'] ? '(' + formData[q.id + '_gem_opp'] + ' m2)' : ''}`, italics: true }));
            if (formData[q.id + '_extra']) children.push(new Paragraph({ text: `Extra: ${formData[q.id + '_extra']}` }));
            if (formData[q.id + '_comment']) children.push(new Paragraph({ text: `Opmerking: ${formData[q.id + '_comment']}`, color: "666666" }));
            if (photoData[q.id]) {
                for (const pic of photoData[q.id]) {
                    const buf = await (await fetch(pic)).arrayBuffer();
                    children.push(new Paragraph({ children: [new ImageRun({ data: buf, transformation: { width: 300, height: 200 } })] }));
                }
            }
        }
        Packer.toBlob(new Document({ sections: [{ children }] })).then(b => saveAs(b, `Opname_${formData['adres'] || 'rapport'}.docx`));
    }// Ondersteuning voor het plakken (paste) van foto's
    window.addEventListener('paste', function(e) {
        // Zoek de kaart waar de gebruiker op dit moment met de muis/cursor is
        const activeElement = document.activeElement;
        const card = activeElement.closest('.card');
        
        if (card) {
            const questionId = card.id.replace('card_', '');
            const items = (e.clipboardData || e.originalEvent.clipboardData).items;
            
            for (const item of items) {
                if (item.kind === 'file' && item.type.includes('image')) {
                    const file = item.getAsFile();
                    processFile(file, questionId); // Hergebruikt de bestaande verwerkingsfunctie
                }
            }
        } else {
            // Optioneel: als er geen specifieke kaart geselecteerd is, 
            // kun je de gebruiker waarschuwen of niets doen.
            console.log("Klik eerst in een vak om een foto te plakken.");
        }
    });
    renderApp();
